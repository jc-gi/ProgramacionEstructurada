using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace CiclojC
{
    class Program
    {
        static void Main(string[] args)
        {
            int i, j, contador = 0;
            for (i = 1; i <= 4; i++)
                for (j = 1; j <= 3; j++)
                    contador = contador + 1;
            Console.WriteLine("El ciclo j se ejecuta {0} veces.", contador);
            Console.ReadLine();
        }
    }
}
